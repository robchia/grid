Code example

<div class="grid">

	<!-- specify column size with l-* -->
	<div class="l-4">
		<!-- specify column with c-* -->
		<div class="c-1">
			<div class="content">column 1</div>
		</div>
		<div class="c-2">
			<div class="content">column 2</div>
		</div>
		<div class="c-3">
			<div class="content">column 3</div>
		</div>
		<div class="c-4">
			<div class="content">column 4</div>
		</div>
		
		<!-- you can group columns with g-* -->
		<div class="c-1 g-3">
			<div class="content">3-column group</div>
		</div>
		<div class="c-2">
			<div class="content">1 column</div>
		</div>
	</div>
